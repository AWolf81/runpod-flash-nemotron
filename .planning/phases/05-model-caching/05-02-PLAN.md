# Cached model configuration — matches nemotron.py CACHED_REPO_ID
CACHED_REPO_ID="your-org/nemotron-q4-xl"  # Set same value as nemotron.py CACHED_REPO_ID

2. Replace the hard-coded `MODEL_PATH=...` assignment with a resolver function:
```bash
resolve_model_path() {
    # Try cached model path first
    if [[ -n "${CACHED_REPO_ID}" ]]; then
        local repo_dir="models--${CACHED_REPO_ID//\/\/--}"
        # Bash-compatible replacement: / -> --
        repo_dir="models--${CACHED_REPO_ID/\//--}"
        local hub_path="${CACHE_BASE}/${repo_dir}"
        local refs_path="${hub_path}/refs/main"
        if [[ -f "${refs_path}" ]]; then
            local hash
            hash="$(cat "${refs_path}")"
            local candidate="${hub_path}/snapshots/${hash}/${MODEL_FILENAME}"
            if [[ -f "${candidate}" ]]; then
                echo "${candidate}"
                return
            fi
        fi
    fi
    # Fall back to network volume path
    echo "/runpod-volume/models/UD-Q4_K_XL/${MODEL_FILENAME}"
}

MODEL_PATH="$(resolve_model_path)"
```

Where `MODEL_FILENAME` is the base filename (extract it from the current hard-coded MODEL_PATH):
```bash
MODEL_FILENAME="NVIDIA-Nemotron-3-Super-120B-A12B-UD-Q4_K_XL-00001-of-00003.gguf"
```

**Note:** The bash string replacement `${CACHED_REPO_ID/\//--}` only replaces the first `/`. The repo ID format is `org/repo` with exactly one slash, so this is correct.
